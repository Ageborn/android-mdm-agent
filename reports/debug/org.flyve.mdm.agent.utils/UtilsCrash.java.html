public class UtilsCrash {

    public static void configCrash(Context context, boolean val) {
        if(val) {
            Bugsnag.init(context);
            Bugsnag.setEndpoint(new EnvInfoSetup(context).getThestralbot());
            Bugsnag.setSendThreads(val);
        } else {
            Bugsnag.init(context);
            Bugsnag.setEndpoint("");
            Bugsnag.setSendThreads(val);
        }

    }
}
